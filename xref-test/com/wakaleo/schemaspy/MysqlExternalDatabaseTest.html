24  
25      /**
26       * Test resources.
27       */
28      @Rule
29      public TestResources resources = new TestResources();
30  
31      /**
32       * test rule.
33       */
34      @Rule
35      public MojoRule rule = new MojoRule();
36  
37      @Rule
38      public TestName name = new TestName();
39  
40      @Test
41      public void testMySqlConfiguration() throws Exception {
42          Logger.getLogger("global").info("Starting: " + name.getMethodName());
43          File projectCopy = this.resources.getBasedir("unit");
44          File testPom = new File(projectCopy,"mysql-plugin-config.xml");
45          assumeNotNull("POM file should not be null.", testPom);
46          assumeTrue("POM file should exist as file.",
47                  testPom.exists() && testPom.isFile());
48  
49          SchemaSpyReport mojo = new SchemaSpyReport();
50          mojo = (SchemaSpyReport) this.rule.configureMojo(mojo, rule.extractPluginConfiguration("schemaspy-maven-plugin", testPom));
51          mojo.executeReport(Locale.getDefault());
52  
53          // check if the reports generated
54          File generatedFile = new File("./target/reports/mysql-test/schemaspy/index.html");
55          Logger.getLogger("global").info("generatedFile = " + generatedFile.getAbsolutePath());
56          assertTrue(generatedFile.exists());
57      }
