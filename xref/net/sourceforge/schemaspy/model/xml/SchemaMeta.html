44  /**
45   * Additional metadata about a schema as expressed in XML instead of from
46   * the database.
47   *
48   * @author John Currier
49   */
50  public class SchemaMeta {
51      private final List<TableMeta> tables = new ArrayList<TableMeta>();
52      private final String comments;
53      private final File metaFile;
54      private final Logger logger = Logger.getLogger(getClass().getName());
55  
56      public SchemaMeta(String xmlMeta, String dbName, String schema) throws InvalidConfigurationException {
57          File meta = new File(xmlMeta);
58          if (meta.isDirectory()) {
59              String filename = (schema == null ? dbName : schema) + ".meta.xml";
60              meta = new File(meta, filename);
61  
62              if (!meta.exists()) {
63                  if (Config.getInstance().isOneOfMultipleSchemas()) {
64                      // don't force all of the "one of many" schemas to have metafiles
65                      logger.info("Meta directory \"" + xmlMeta + "\" should contain a file named \"" + filename + '\"');
66                      comments = null;
67                      metaFile = null;
68                      return;
69                  }
70  
71                  throw new InvalidConfigurationException("Meta directory \"" + xmlMeta + "\" must contain a file named \"" + filename + '\"');
72              }
73          } else if (!meta.exists()) {
74              throw new InvalidConfigurationException("Specified meta file \"" + xmlMeta + "\" does not exist");
75          }
76  
77          metaFile = meta;
78  
79          Document doc = parse(metaFile);
80  
81          NodeList commentsNodes = doc.getElementsByTagName("comments");
82          if (commentsNodes != null && commentsNodes.getLength() > 0)
83              comments = commentsNodes.item(0).getTextContent();
84          else
85              comments = null;
86  
87          NodeList tablesNodes = doc.getElementsByTagName("tables");
88          if (tablesNodes != null) {
89              NodeList tableNodes = ((Element)tablesNodes.item(0)).getElementsByTagName("table");
90  
91              for (int i = 0; i < tableNodes.getLength(); ++i) {
92                  Node tableNode = tableNodes.item(i);
93                  TableMeta/xml/TableMeta.html#TableMeta">TableMeta tableMeta = new TableMeta(tableNode);
94                  tables.add(tableMeta);
95              }
96          }
97      }

111         return tables;

114     private void validate(Document document) throws SAXException, IOException {
115         // create a SchemaFactory capable of understanding WXS schemas
116         SchemaFactory factory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
117 
118         // load a WXS schema, represented by a Schema instance
119         InputStream xsl = getClass().getResourceAsStream("/schemaspy.meta.xsd");
120 
121         Schema schema = factory.newSchema(new StreamSource(xsl));
122 
123         // create a Validator instance, which can be used to validate an instance document
124         Validator validator = schema.newValidator();
125 
126         // validate the DOM tree
127         validator.validate(new DOMSource(document));
128     }
129 
130     private Document parse(File file) throws InvalidConfigurationException {
131         DocumentBuilder docBuilder;
132         Document doc = null;
133         DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
134         factory.setNamespaceAware(true);
135         factory.setIgnoringElementContentWhitespace(true);
136         factory.setIgnoringComments(true);
137 
138         try {
139             docBuilder = factory.newDocumentBuilder();
140         } catch (ParserConfigurationException exc) {
141             throw new InvalidConfigurationException("Invalid XML parser configuration", exc);
142         }
143 
144         try {
145             doc = docBuilder.parse(file);
146             validate(doc);
147         } catch (SAXException exc) {
148             throw new InvalidConfigurationException(file + " failed XML validation:", exc);
149         } catch (IOException exc) {
150             throw new InvalidConfigurationException("Could not read " + file + ":", exc);
151         }
152 
153         return doc;
154     }
