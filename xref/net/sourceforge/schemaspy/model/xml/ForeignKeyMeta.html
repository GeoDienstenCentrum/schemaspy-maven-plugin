19  package net.sourceforge.schemaspy.model.xml;
20  
21  import java.util.logging.Logger;
22  import org.w3c.dom.NamedNodeMap;
23  import org.w3c.dom.Node;
24  
25  /**
26   * Additional metadata about a foreign key relationship as expressed in XML
27   * instead of from the database.
28   *
29   * @author John Currier
30   */
31  public class ForeignKeyMeta {
32      private final String tableName;
33      private final String columnName;
34      private final String remoteSchema;
35      private final static Logger logger = Logger.getLogger(ForeignKeyMeta.class.getName());
36  
37      ForeignKeyMeta(Node foreignKeyNode) {
38          NamedNodeMap attribs = foreignKeyNode.getAttributes();
39          Node node = attribs.getNamedItem("table");
40          if (node == null)
41              throw new IllegalStateException("XML foreignKey definition requires 'table' attribute");
42          tableName = node.getNodeValue();
43          node = attribs.getNamedItem("column");
44          if (node == null)
45              throw new IllegalStateException("XML foreignKey definition requires 'column' attribute");
46          columnName = node.getNodeValue();
47          node = attribs.getNamedItem("remoteSchema");
48          if (node != null) {
49              remoteSchema = node.getNodeValue();
50          } else {
51              remoteSchema = null;
52          }
53  
54          logger.finer("Found XML FK metadata for " + tableName + "." + columnName +
55                  " remoteSchema: " + remoteSchema);
56      }
57  
58      public String getTableName() {
59          return tableName;
60      }
61  
62      public String getColumnName() {
63          return columnName;
64      }
65  
66      public String getRemoteSchema() {
67          return remoteSchema;
68      }
69  
70      @Override
71      public String toString() {
72          return tableName + '.' + columnName;
73      }
74  }
