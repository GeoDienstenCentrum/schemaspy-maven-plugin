19  package net.sourceforge.schemaspy;
20  
21  import net.sourceforge.schemaspy.model.ConnectionFailure;
22  import net.sourceforge.schemaspy.model.EmptySchemaException;
23  import net.sourceforge.schemaspy.model.InvalidConfigurationException;
24  import net.sourceforge.schemaspy.model.ProcessExecutionException;
25  import net.sourceforge.schemaspy.ui.MainFrame;
26  
27  /**
28   * @author John Currier
29   */
30  public class Main {
31      public static void main(String[] argv) throws Exception {
32          if (argv.length == 1 && argv[0].equals("-gui")) { // warning: serious temp hack
33              new MainFrame().setVisible(true);
34              return;
35          }
36  
37          SchemaAnalyzer analyzer = new SchemaAnalyzer();
38  
39          int rc = 1;
40  
41          try {
42              rc = analyzer.analyze(new Config(argv)) == null ? 1 : 0;
43          } catch (ConnectionFailure couldntConnect) {
44              // failure already logged
45              rc = 3;
46          } catch (EmptySchemaException noData) {
47              // failure already logged
48              rc = 2;
49          } catch (InvalidConfigurationException badConfig) {
50              System.err.println();
51              if (badConfig.getParamName() != null)
52                  System.err.println("Bad parameter specified for " + badConfig.getParamName());
53              System.err.println(badConfig.getMessage());
54              if (badConfig.getCause() != null && !badConfig.getMessage().endsWith(badConfig.getMessage()))
55                  System.err.println(" caused by " + badConfig.getCause().getMessage());
56          } catch (ProcessExecutionException badLaunch) {
57              System.err.println(badLaunch.getMessage());
58          } catch (Exception exc) {
59              exc.printStackTrace();
60          }
61  
62          System.exit(rc);
63      }
64  }
