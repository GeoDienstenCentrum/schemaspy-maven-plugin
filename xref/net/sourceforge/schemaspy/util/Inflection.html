106     private final String pattern;
107     private final String replacement;
108     private final boolean ignoreCase;
109 
110     public Inflection(String pattern) {
111         this(pattern, null, true);
112     }
113 
114     public Inflection(String pattern, String replacement) {
115         this(pattern, replacement, true);
116     }
117 
118     public Inflection(String pattern, String replacement, boolean ignoreCase) {
119         this.pattern = pattern;
120         this.replacement = replacement;
121         this.ignoreCase = ignoreCase;
122     }
123 
124     private static void plural(String pattern, String replacement) {
125         plural.add(0, new Inflection(pattern, replacement));
126     }
127 
128     private static void singular(String pattern, String replacement) {
129         singular.add(0, new Inflection(pattern, replacement));
130     }
131 
132     private static void irregular(String s, String p) {
133         plural("(" + s.substring(0, 1) + ")" + s.substring(1) + "$", "$1" + p.substring(1));
134         singular("(" + p.substring(0, 1) + ")" + p.substring(1) + "$", "$1" + s.substring(1));
135     }
136 
137     private static void uncountable(String word) {
138         uncountable.add(word);
139     }
140 
141     /**
142      * Does the given word match?
143      * @param word The word
144      * @return True if it matches the inflection pattern
145      */
146     public boolean match(String word) {
147         int flags = 0;
148         if (ignoreCase) {
149             flags = flags | Pattern.CASE_INSENSITIVE;
150         }
151         return Pattern.compile(pattern, flags).matcher(word).find();
152     }
153 
154     /**
155      * Replace the word with its pattern.
156      * @param word The word
157      * @return The result
158      */
159     public String replace(String word) {
160         int flags = 0;
161         if (ignoreCase) {
162             flags = flags | Pattern.CASE_INSENSITIVE;
163         }
164         return Pattern.compile(pattern, flags).matcher(word).replaceAll(replacement);
165     }
166 
167     /**
168      * Return the pluralized version of a word.
169      * @param word The word
170      * @return The pluralized word
171      */
172     public static String pluralize(String word) {
173         if (Inflection.isUncountable(word)) {
174             return word;
175         }
176 
177         for (Inflection inflection : plural) {
178             if (inflection.match(word)) {
179                 return inflection.replace(word);
180             }
181         }
182         return word;
183     }
184 
185     /**
186      * Return the singularized version of a word.
187      * @param word The word
188      * @return The singularized word
189      */
190     public static String singularize(String word) {
191         if (Inflection.isUncountable(word)) {
192             return word;
193         }
194 
195         for (Inflection inflection : singular) {
196             //System.out.println(word + " matches " + inflection.pattern + "? (ignore case: " + inflection.ignoreCase + ")");
197             if (inflection.match(word)) {
198                 //System.out.println("match!");
199                 return inflection.replace(word);
200             }
201         }
202         return word;
203     }
204 
205     /**
206      * Return true if the word is uncountable.
207      * @param word The word
208      * @return True if it is uncountable
209      */
210     public static boolean isUncountable(String word) {
211         for (String w : uncountable) {
212             if (w.equalsIgnoreCase(word)) return true;
213         }
214         return false;
215     }
