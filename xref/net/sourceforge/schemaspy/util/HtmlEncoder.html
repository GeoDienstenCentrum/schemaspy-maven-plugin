19  package net.sourceforge.schemaspy.util;
20  
21  import java.util.HashMap;
22  import java.util.Map;
23  
24  /**
25   * Simple (i.e. 'stupid') class that does a simple mapping between
26   * HTML characters and their 'encoded' equivalents.
27   *
28   * @author John Currier
29   */
30  public class HtmlEncoder {
31      private static final Map<String, String> map = new HashMap<String, String>();
32  
33      static {
34          map.put("<", "&lt;");
35          map.put(">", "&gt;");
36          map.put("\n", "<br>" + System.getProperty("line.separator"));
37          map.put("\r", "");
38      }
39  
40      private HtmlEncoder() {}
41  
42      /**
43       * Returns an HTML-encoded equivalent of the specified character.
44       *
45       * @param ch
46       * @return
47       */
48      public static String encodeToken(char ch) {
49          return encodeToken(String.valueOf(ch));
50      }
51  
52      /**
53       * Returns an HTML-encoded equivalent of the specified tokenized string,
54       * where tokens such as '<', '>', '\n' and '\r' have been isolated from
55       * other tokens.
56       *
57       * @param str
58       * @return
59       */
60      public static String encodeToken(String str) {
61          String result = map.get(str);
62          return (result == null) ? str : result;
63      }
64  
65      /**
66       * Returns an HTML-encoded version of the specified string
67       *
68       * @param str
69       * @return
70       */
71      public static String encodeString(String str) {
72          int len = str.length();
73      	StringBuilder buf = new StringBuilder(len * 2); // x2 should limit # of reallocs
74      	for (int i = 0; i < len; i++) {
75  			buf.append(encodeToken(str.charAt(i)));
76  		}
77      	return buf.toString();
78      }
79  }
