19  package net.sourceforge.schemaspy.view;
20  
21  import java.util.Set;
22  import net.sourceforge.schemaspy.model.Database;
23  import net.sourceforge.schemaspy.model.Table;
24  
25  /**
26   * Implementations of this interface know how to take SQL and format it
27   * into (hopefully) readable HTML.
28   *
29   * @author John Currier
30   */
31  public interface SqlFormatter {
32      /**
33       * Return a HTML-formatted representation of the specified SQL.
34       *
35       * @param sql SQL to be formatted
36       * @param db Database
37       * @param references set of tables referenced by this SQL
38       *      (populated by the formatter) or left empty if the formatter already
39       *      provides references to those tables
40       * @return HTML-formatted representation of the specified SQL
41       */
42      String format(String sql, Database db, Set<Table> references);
43  }
