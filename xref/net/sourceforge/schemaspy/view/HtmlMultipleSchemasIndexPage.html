35  public class HtmlMultipleSchemasIndexPage extends HtmlFormatter {
36      private static HtmlMultipleSchemasIndexPagedexPage.html#HtmlMultipleSchemasIndexPage">HtmlMultipleSchemasIndexPage instance = new HtmlMultipleSchemasIndexPage();

42      }

53      public void write(String dbName, List<String> populatedSchemas, DatabaseMetaData meta, LineWriter index) throws IOException {
54          writeHeader(dbName, meta, populatedSchemas.size(), false, populatedSchemas.get(0).toString(), index);
55  
56          for (String schema : populatedSchemas) {
57              writeLineItem(schema, index);
58          }
59  
60          writeFooter(index);
61      }
62  
63      private void writeHeader(String databaseName, DatabaseMetaData meta, int numberOfSchemas, boolean showIds, String aSchema, LineWriter html) throws IOException {
64          String connectTime = new SimpleDateFormat("EEE MMM dd HH:mm z yyyy").format(new Date());
65  
66          html.writeln("<!DOCTYPE HTML PUBLIC '-//W3C//DTD HTML 4.01 Transitional//EN' 'http://www.w3.org/TR/html4/loose.dtd'>");
67          html.writeln("<html>");
68          html.writeln("<head>");
69          html.write("  <title>SchemaSpy Analysis");
70          if (databaseName != null) {
71              html.write(" of Database ");
72              html.write(databaseName);
73          }
74          html.writeln("</title>");
75          html.write("  <link rel=stylesheet href='");
76          html.write(encodeHref(aSchema));
77          html.writeln("/schemaSpy.css' type='text/css'>");
78          html.writeln("  <meta HTTP-EQUIV='Content-Type' CONTENT='text/html; charset=" + Config.getInstance().getCharset() + "'>");
79          html.writeln("</head>");
80          html.writeln("<body>");
81          writeTableOfContents(html);
82          html.writeln("<div class='content' style='clear:both;'>");
83          html.writeln("<table width='100%' border='0' cellpadding='0'>");
84          html.writeln(" <tr>");
85          html.write("  <td class='heading' valign='top'><h1>");
86          html.write("SchemaSpy Analysis");
87          if (databaseName != null) {
88              html.write(" of Database ");
89              html.write(databaseName);
90          }
91          html.writeln("</h1></td>");
92          html.writeln("  <td class='heading' align='right' valign='top' title='John Currier - Creator of Cool Tools'><span class='indent'>Generated by</span><br><span class='indent'><span class='signature'><a href='http://schemaspy.sourceforge.net' target='_blank'>SchemaSpy</a></span></span></td>");
93          html.writeln(" </tr>");
94          html.writeln("</table>");
95          html.writeln("<table width='100%'>");
96          html.writeln(" <tr><td class='container'>");
97          writeGeneratedBy(connectTime, html);
98          html.writeln(" </td></tr>");
99          html.writeln(" <tr>");
100         html.write("  <td class='container'>");
101         if (meta != null) {
102             html.write("Database Type: ");
103             html.write(getDatabaseProduct(meta));
104         }
105         html.writeln("  </td>");
106         html.writeln("  <td class='container' align='right' valign='top' rowspan='3'>");
107         if (sourceForgeLogoEnabled())
108             html.writeln("    <a href='http://sourceforge.net' target='_blank'><img src='http://sourceforge.net/sflogo.php?group_id=137197&amp;type=1' alt='SourceForge.net' border='0' height='31' width='88'></a><br>");
109         html.write("    <br>");
110         writeFeedMe(html);
111         html.writeln("  </td>");
112         html.writeln(" </tr>");
113         html.writeln("</table>");
114 
115         html.writeln("<div class='indent'>");
116         html.write("<b>");
117         html.write(String.valueOf(numberOfSchemas));
118         if (databaseName != null)
119             html.write(" Schema");
120         else
121             html.write(" Database");
122         html.write(numberOfSchemas == 1 ? "" : "s");
123         html.writeln(":</b>");
124         html.writeln("<TABLE class='dataTable' border='1' rules='groups'>");
125         html.writeln("<colgroup>");
126         html.writeln("<thead align='left'>");
127         html.writeln("<tr>");
128         html.write("  <th valign='bottom'>");
129         if (databaseName != null)
130             html.write("Schema");
131         else
132             html.write("Database");
133         html.writeln("</th>");
134         if (showIds)
135             html.writeln("  <th align='center' valign='bottom'>ID</th>");
136         html.writeln("</tr>");
137         html.writeln("</thead>");
138         html.writeln("<tbody>");
139     }

141     private void writeLineItem(String schema, LineWriter index) throws IOException {
142         index.writeln(" <tr>");
143         index.write("  <td class='detail'><a href='");
144         index.write(encodeHref(schema));
145         index.write("/index.html'>");
146         index.write(schema);
147         index.writeln("</a></td>");
148         index.writeln(" </tr>");
149     }
150 
151     protected void writeTableOfContents(LineWriter html) throws IOException {
152         // have to use a table to deal with a horizontal scrollbar showing up inappropriately
153         html.writeln("<table id='headerHolder' cellspacing='0' cellpadding='0'><tr><td>");
154         html.writeln("<div id='header'>");
155         html.writeln(" <ul>");
156         html.writeln("  <li id='current'><a href='index.html' title='All user schemas in the database'>Schemas</a></li>");
157         html.writeln("  <li><a href='http://sourceforge.net/donate/index.php?group_id=137197' title='Please help keep SchemaSpy alive' target='_blank'>Donate</a></li>");
158         html.writeln(" </ul>");
159         html.writeln("</div>");
160         html.writeln("</td></tr></table>");
161     }
162 
163     /**
164      * Copy / paste from Database, but we can't use Database here...
165      *
166      * @param meta DatabaseMetaData
167      * @return String
168      */
169     private String getDatabaseProduct(DatabaseMetaData meta) {
170         try {
171             return meta.getDatabaseProductName() + " - " + meta.getDatabaseProductVersion();
172         } catch (SQLException exc) {
173             return "";
174         }
175     }
