58      public void write(Database database, Collection<Table> tables, boolean showOrphansDiagram, LineWriter html) throws IOException {
59          Set<Table> byName = new TreeSet<Table>(new Comparator<Table>() {
60              public int compare(Table table1, Table table2) {
61                  return table1.compareTo(table2);
62              }
63          });
64          byName.addAll(tables);
65  
66          boolean showIds = false;
67          int numViews = 0;
68          boolean comments = false;
69  
70          for (Table table : byName) {
71              if (table.isView())
72                  ++numViews;
73              showIds |= table.getId() != null;
74              if (table.getComments() != null)
75                  comments = true;
76          }
77  
78          writeHeader(database, byName.size() - numViews, numViews, showIds, showOrphansDiagram, comments, html);
79  
80          int numTableCols = 0;
81          int numViewCols = 0;
82          long numRows = 0;
83          for (Table table : byName) {
84              writeLineItem(table, showIds, html);
85  
86              if (!table.isView())
87                  numTableCols += table.getColumns().size();
88              else
89                  numViewCols += table.getColumns().size();
90              numRows += table.getNumRows();
91          }
92  
93          writeFooter(byName.size() - numViews, numTableCols, numViews, numViewCols, numRows, html);
94      }

96      private void writeHeader(Database db, int numberOfTables, int numberOfViews, boolean showIds, boolean hasOrphans, boolean hasComments, LineWriter html) throws IOException {
97          List<String> javascript = new ArrayList<String>();
98  
99          // we can't use the hard-coded even odd technique that we use
100         // everywhere else because we're dynamically changing the visibility
101         // of tables/views within the list
102         javascript.add("$(function(){");
103         javascript.add("  associate($('#showTables'), $('.tbl'));");
104         javascript.add("  associate($('#showViews'),  $('.view'));");
105         javascript.add("  jQuery.fn.alternateRowColors = function() {");
106         javascript.add("    $('tbody tr:visible').each(function(i) {");
107         javascript.add("      if (i % 2 == 0) {");
108         javascript.add("        $(this).removeClass('even').addClass('odd');");
109         javascript.add("      } else {");
110         javascript.add("        $(this).removeClass('odd').addClass('even');");
111         javascript.add("      }");
112         javascript.add("    });");
113         javascript.add("    return this;");
114         javascript.add("  };");
115         javascript.add("  $('#showTables, #showViews').click(function() {");
116         javascript.add("    $('table.dataTable').alternateRowColors();");
117         javascript.add("  });");
118         javascript.add("  $('table.dataTable').alternateRowColors();");
119         javascript.add("})");
120 
121         writeHeader(db, null, null, hasOrphans, javascript, html);
122         html.writeln("<table width='100%'>");
123         html.writeln(" <tr><td class='container'>");
124         writeGeneratedBy(db.getConnectTime(), html);
125         html.writeln(" </td></tr>");
126         html.writeln(" <tr>");
127         html.write("  <td class='container'>Database Type: ");
128         html.write(db.getDatabaseProduct());
129         html.writeln("  </td>");
130         html.writeln("  <td class='container' align='right' valign='top' rowspan='3'>");
131         if (sourceForgeLogoEnabled())
132             html.writeln("    <a href='http://sourceforge.net' target='_blank'><img src='http://sourceforge.net/sflogo.php?group_id=137197&amp;type=1' alt='SourceForge.net' border='0' height='31' width='88'></a><br>");
133         html.write("    <br>");
134         writeFeedMe(html);
135         html.writeln("  </td>");
136         html.writeln(" </tr>");
137         html.writeln(" <tr>");
138         html.write("  <td class='container'>");
139         String xmlName = db.getName();
140         if (db.getSchema() != null)
141             xmlName += '.' + db.getSchema();
142         html.write("<br><a href='" + xmlName + ".xml' title='XML Representation'>XML Representation</a>");
143         html.write("<br><a href='insertionOrder.txt' title='Useful for loading data into a database'>Insertion Order</a>&nbsp;");
144         html.write("<a href='deletionOrder.txt' title='Useful for purging data from a database'>Deletion Order</a>");
145         html.write("&nbsp;(for database loading/purging scripts)");
146         html.writeln("</td>");
147         html.writeln(" </tr>");
148         html.writeln("</table>");
149 
150         html.writeln("<div class='indent'>");
151         html.write("<p>");
152         html.write("<b>");
153         if (numberOfViews == 0) {
154             html.writeln("<label for='showTables' style='display:none;'><input type='checkbox' id='showTables' checked></label>");
155         } else if (numberOfTables == 0) {
156             html.writeln("<label for='showViews' style='display:none;'><input type='checkbox' id='showViews' checked></label>");
157         } else {
158             html.write("<label for='showTables'><input type='checkbox' id='showTables' checked>Tables</label>");
159             html.write(" <label for='showViews'><input type='checkbox' id='showViews' checked>Views</label>");
160         }
161 
162         html.writeln(" <label for='showComments'><input type=checkbox " + (hasComments  ? "checked " : "") + "id='showComments'>Comments</label>");
163         html.writeln("</b>");
164 
165         html.writeln("<table class='dataTable' border='1' rules='groups'>");
166         int numGroups = 4 + (showIds ? 1 : 0) + (displayNumRows ? 1 : 0);
167         for (int i = 0; i < numGroups; ++i)
168             html.writeln("<colgroup>");
169         html.writeln("<colgroup class='comment'>");
170         html.writeln("<thead align='left'>");
171         html.writeln("<tr>");
172         String tableHeading;
173         if (numberOfViews == 0)
174             tableHeading = "Table";
175         else if (numberOfTables == 0)
176             tableHeading = "View";
177         else
178             tableHeading = "Table / View";
179         html.writeln("  <th valign='bottom'>" + tableHeading + "</th>");
180         if (showIds)
181             html.writeln("  <th align='center' valign='bottom'>ID</th>");
182         html.writeln("  <th align='right' valign='bottom'>Children</th>");
183         html.writeln("  <th align='right' valign='bottom'>Parents</th>");
184         html.writeln("  <th align='right' valign='bottom'>Columns</th>");
185         if (displayNumRows)
186             html.writeln("  <th align='right' valign='bottom'>Rows</th>");
187         html.writeln("  <th class='comment' align='left' valign='bottom'>Comments</th>");
188         html.writeln("</tr>");
189         html.writeln("</thead>");
190         html.writeln("<tbody>");
191     }

193     private void writeLineItem(Table table, boolean showIds, LineWriter html) throws IOException {
194         html.write(" <tr class='" + (table.isView() ? "view" : "tbl") + "' valign='top'>");
195         html.write("  <td class='detail'><a href='tables/");
196         html.write(encodeHref(table.getName()));
197         html.write(".html'>");
198         html.write(table.getName());
199         html.writeln("</a></td>");
200 
201         if (showIds) {
202             html.write("  <td class='detail' align='right'>");
203             Object id = table.getId();
204             if (id != null)
205                 html.write(String.valueOf(id));
206             else
207                 html.writeln("&nbsp;");
208             html.writeln("</td>");
209         }
210 
211         html.write("  <td class='detail' align='right'>");
212         int numRelatives = table.getNumNonImpliedChildren();
213         if (numRelatives != 0)
214             html.write(String.valueOf(integerFormatter.format(numRelatives)));
215         html.writeln("</td>");
216         html.write("  <td class='detail' align='right'>");
217         numRelatives = table.getNumNonImpliedParents();
218         if (numRelatives != 0)
219             html.write(String.valueOf(integerFormatter.format(numRelatives)));
220         html.writeln("</td>");
221 
222         html.write("  <td class='detail' align='right'>");
223         html.write(String.valueOf(integerFormatter.format(table.getColumns().size())));
224         html.writeln("</td>");
225 
226         if (displayNumRows) {
227             html.write("  <td class='detail' align='right'>");
228             if (!table.isView())
229                 html.write(String.valueOf(integerFormatter.format(table.getNumRows())));
230             else
231                 html.write("<span title='Views contain no real rows'>view</span>");
232             html.writeln("</td>");
233         }
234         html.write("  <td class='comment detail'>");
235         String comments = table.getComments();
236         if (comments != null) {
237             if (encodeComments)
238                 for (int i = 0; i < comments.length(); ++i)
239                     html.write(HtmlEncoder.encodeToken(comments.charAt(i)));
240             else
241                 html.write(comments);
242         }
243         html.writeln("</td>");
244         html.writeln("  </tr>");
245     }

247     protected void writeFooter(int numTables, int numTableCols, int numViews, int numViewCols, long numRows, LineWriter html) throws IOException {
248         html.writeln("  <tr>");
249         html.writeln("    <td class='detail'>&nbsp;</td>");
250         html.writeln("    <td class='detail'>&nbsp;</td>");
251         html.writeln("    <td class='detail'>&nbsp;</td>");
252         html.writeln("    <td class='detail'>&nbsp;</td>");
253         if (displayNumRows)
254             html.writeln("    <td class='detail'>&nbsp;</td>");
255         html.writeln("    <td class='comment detail'>&nbsp;</td>");
256         html.writeln("  </tr>");
257         String name = numTables == 1 ? " Table" : " Tables";
258         html.writeln("  <tr class='tbl'>");
259         html.writeln("    <td class='detail'><b>" + integerFormatter.format(numTables) + name + "</b></td>");
260         html.writeln("    <td class='detail'>&nbsp;</td>");
261         html.writeln("    <td class='detail'>&nbsp;</td>");
262         html.writeln("    <td class='detail' align='right'><b>" + integerFormatter.format(numTableCols) + "</b></td>");
263         if (displayNumRows)
264             html.writeln("    <td class='detail' align='right'><b>" + integerFormatter.format(numRows) + "</b></td>");
265         html.writeln("    <td class='comment detail'>&nbsp;</td>");
266         html.writeln("  </tr>");
267         name = numViews == 1 ? " View" : " Views";
268         html.writeln("  <tr class='view'>");
269         html.writeln("    <td class='detail'><b>" + integerFormatter.format(numViews) + name + "</b></td>");
270         html.writeln("    <td class='detail'>&nbsp;</td>");
271         html.writeln("    <td class='detail'>&nbsp;</td>");
272         html.writeln("    <td class='detail' align='right'><b>" + integerFormatter.format(numViewCols) + "</b></td>");
273         if (displayNumRows)
274             html.writeln("    <td class='detail'>&nbsp;</td>");
275         html.writeln("    <td class='comment detail'>&nbsp;</td>");
276         html.writeln("  </tr>");
277         html.writeln("</table>");
278 
279         super.writeFooter(html);
280     }
