24  import java.util.Set;

31  public class DotNode {
32      private final Table table;
33      private final DotNodeConfig config;
34      private final String path;
35      private final Set<TableColumn> excludedColumns = new HashSet<TableColumn>();
36      private final String lineSeparator = System.getProperty("line.separator");
37      private final boolean displayNumRows = Config.getInstance().isNumRowsEnabled();
38      private static final URLEncoder urlEncoder = new URLEncoder(Config.DOT_CHARSET);
39  
40      /**
41       * Create a DotNode that is a focal point of a diagram.
42       * That is, all of its columns are displayed in addition to the details
43       * of those columns.
44       *
45       * @param table Table
46       * @param path String
47       */
48      public DotNode(Table table, String path) {
49          this(table, path, new DotNodeConfig(true, true));
50      }
51  
52      public DotNode(Table table, String path, DotNodeConfig config) {
53          this.table = table;
54          this.path = path + (table.isRemote() ? ("../../" + table.getSchema() + "/tables/") : "");
55          this.config = config;
56      }
57  
58      /**
59       * Create a DotNode and specify whether it displays its columns.
60       * The details of the optional columns (e.g. type, size) are not displayed.
61       *
62       * @param table Table
63       * @param showColumns boolean
64       * @param path String
65       */
66      public DotNode(Table table, boolean showColumns, String path) {
67          this(table, path, showColumns ? new DotNodeConfig(true, false) : new DotNodeConfig());
68      }
69  
70      public void setShowImplied(boolean showImplied) {
71          config.showImpliedRelationships = showImplied;
72      }
73  
74      public Table getTable() {
75          return table;
76      }
77  
78      public void excludeColumn(TableColumn column) {
79          excludedColumns.add(column);
80      }
81  
82      @Override
83      public String toString() {
84          StyleSheet css = StyleSheet.getInstance();
85          StringBuilder buf = new StringBuilder();
86          String tableName = table.getName();
87          // fully qualified table name (optionally prefixed with schema)
88          String fqTableName = (table.isRemote() ? table.getSchema() + "." : "") + tableName;
89          String colspan = config.showColumnDetails ? "COLSPAN=\"2\" " : "COLSPAN=\"3\" ";
90  
91          buf.append("  \"" + fqTableName + "\" [" + lineSeparator);
92          buf.append("    label=<" + lineSeparator);
93          buf.append("    <TABLE BORDER=\"" + (config.showColumnDetails ? "2" : "0") + "\" CELLBORDER=\"1\" CELLSPACING=\"0\" BGCOLOR=\"" + css.getTableBackground() + "\">" + lineSeparator);
94          buf.append("      <TR>");
95          buf.append("<TD COLSPAN=\"3\" BGCOLOR=\"" + css.getTableHeadBackground() + "\" ALIGN=\"CENTER\">" + fqTableName + "</TD>");
96          buf.append("</TR>" + lineSeparator);
97  
98          boolean skippedTrivial = false;
99  
100         if (config.showColumns) {
101             List<TableColumn> primaryColumns = table.getPrimaryColumns();
102             Set<TableColumn> indexColumns = new HashSet<TableColumn>();
103 
104             for (TableIndex index : table.getIndexes()) {
105                 indexColumns.addAll(index.getColumns());
106             }
107             indexColumns.removeAll(primaryColumns);
108 
109             for (TableColumn column : table.getColumns()) {
110                 if (config.showTrivialColumns || config.showColumnDetails || column.isPrimary() || column.isForeignKey() || indexColumns.contains(column)) {
111                     buf.append("      <TR>");
112                     buf.append("<TD PORT=\"" + column.getName() + "\" " + colspan);
113                     if (excludedColumns.contains(column))
114                         buf.append("BGCOLOR=\"" + css.getExcludedColumnBackgroundColor() + "\" ");
115                     else if (primaryColumns.contains(column))
116                         buf.append("BGCOLOR=\"" + css.getPrimaryKeyBackground() + "\" ");
117                     else if (indexColumns.contains(column))
118                         buf.append("BGCOLOR=\"" + css.getIndexedColumnBackground() + "\" ");
119                     buf.append("ALIGN=\"LEFT\">");
120                     buf.append(column.getName());
121                     buf.append("</TD>");
122                     if (config.showColumnDetails) {
123                         buf.append("<TD PORT=\"");
124                         buf.append(column.getName());
125                         buf.append(".type\" ALIGN=\"LEFT\">");
126                         buf.append(column.getType().toLowerCase());
127                         buf.append("[");
128                         buf.append(column.getDetailedSize());
129                         buf.append("]</TD>");
130                     }
131                     buf.append("</TR>" + lineSeparator);
132                 } else {
133                     skippedTrivial = true;
134                 }
135             }
136         }
137 
138         if (skippedTrivial || !config.showColumns) {
139             buf.append("      <TR><TD PORT=\"elipses\" COLSPAN=\"3\" ALIGN=\"LEFT\">...</TD></TR>" + lineSeparator);
140         }
141 
142         buf.append("      <TR>");
143         buf.append("<TD ALIGN=\"LEFT\" BGCOLOR=\"" + css.getBodyBackground() + "\">");
144         int numParents = config.showImpliedRelationships ? table.getNumParents() : table.getNumNonImpliedParents();
145         if (numParents > 0 || config.showColumnDetails)
146             buf.append("&lt; " + numParents);
147         else
148             buf.append("  ");
149         buf.append("</TD>");
150         buf.append("<TD ALIGN=\"RIGHT\" BGCOLOR=\"" + css.getBodyBackground() + "\">");
151         if (table.isView())
152             buf.append("view");
153         else {
154             final long numRows = table.getNumRows();
155             if (displayNumRows && numRows != -1) {
156                 buf.append(NumberFormat.getInstance().format(numRows));
157                 buf.append(" row");
158                 if (numRows != 1)
159                     buf.append('s');
160             } else {
161                 buf.append("  ");
162             }
163         }
164         buf.append("</TD>");
165         buf.append("<TD ALIGN=\"RIGHT\" BGCOLOR=\"" + css.getBodyBackground() + "\">");
166         int numChildren = config.showImpliedRelationships ? table.getNumChildren() : table.getNumNonImpliedChildren();
167         if (numChildren > 0 || config.showColumnDetails)
168             buf.append(numChildren + " &gt;");
169         else
170             buf.append("  ");
171         buf.append("</TD></TR>" + lineSeparator);
172 
173         buf.append("    </TABLE>>" + lineSeparator);
174         if (!table.isRemote() || Config.getInstance().isOneOfMultipleSchemas())
175             buf.append("    URL=\"" + path + toNCR( urlEncoder.encode(tableName) ) + ".html\"" + lineSeparator);
176         buf.append("    tooltip=\"" + toNCR(fqTableName) + "\"" + lineSeparator);
177         buf.append("  ];");
178 
179         return buf.toString();
180     }

182     /**
183      * Translates specified string to Numeric Character Reference (NCR).
184      * This (hopefully) allows Unicode languages to be displayed correctly.<p>
185      * The basis for this code was found
186      * <a href='http://d.hatena.ne.jp/etherealmaro/20060806#1154886500'>here</a>.
187      *
188      * @param str
189      * @return
190      */
191     private static String toNCR(String str) {
192         StringBuilder result = new StringBuilder();
193         for (int i = 0; i < str.length(); ++i) {
194             char ch = str.charAt(i);
195             if (ch <= 127) {    // don't confuse things unless necessary
196                 result.append(ch);
197             } else {
198                 result.append("&#");
199                 result.append(Integer.parseInt(Integer.toHexString(ch), 16));
200                 result.append(";");
201             }
202         }
203         return result.toString();
204     }

206     public static class DotNodeConfig {
207         private final boolean showColumns;
208         private boolean showTrivialColumns;
209         private final boolean showColumnDetails;
210         private boolean showImpliedRelationships;
211 
212         /**
213          * Nothing but table name and counts are displayed
214          */
215         public DotNodeConfig() {
216             showColumns = showTrivialColumns = showColumnDetails = showImpliedRelationships = false;
217         }
218 
219         public DotNodeConfig(boolean showTrivialColumns, boolean showColumnDetails) {
220             showColumns = true;
221             this.showTrivialColumns = showTrivialColumns;
222             this.showColumnDetails = showColumnDetails;
223         }
224     }
