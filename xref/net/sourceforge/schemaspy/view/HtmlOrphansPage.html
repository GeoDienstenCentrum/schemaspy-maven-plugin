55      public boolean write(Database db, List<Table> orphanTables, File diagramDir, LineWriter html) throws IOException {
56          Dot dot = getDot();
57          if (dot == null)
58              return false;
59  
60          Set<Table> orphansWithImpliedRelationships = new HashSet<Table>();
61  
62          for (Table table : orphanTables) {
63              if (!table.isOrphan(true)){
64                  orphansWithImpliedRelationships.add(table);
65              }
66          }
67  
68          writeHeader(db, "Utility Tables", !orphansWithImpliedRelationships.isEmpty(), html);
69  
70          html.writeln("<a name='diagram'>");
71          try {
72              StringBuilder maps = new StringBuilder(64 * 1024);
73  
74              for (Table table : orphanTables) {
75                  String dotBaseFilespec = table.getName();
76  
77                  File dotFile = new File(diagramDir, dotBaseFilespec + ".1degree.dot");
78                  File imgFile = new File(diagramDir, dotBaseFilespec + ".1degree.png");
79  
80                  LineWriterineWriter.html#LineWriter">LineWriter dotOut = new LineWriter(dotFile, Config.DOT_CHARSET);
81                  DotFormatter.getInstance().writeOrphan(table, dotOut);
82                  dotOut.close();
83                  try {
84                      maps.append(dot.generateDiagram(dotFile, imgFile));
85                  } catch (Dot.DotFailure dotFailure) {
86                      System.err.println(dotFailure);
87                      return false;
88                  }
89  
90                  html.write("  <img src='diagrams/summary/" + encodeHref( imgFile.getName() ) + "' usemap='#" + table + "' border='0' alt='' align='top'");
91                  if (orphansWithImpliedRelationships.contains(table))
92                      html.write(" class='impliedNotOrphan'");
93                  html.writeln(">");
94              }
95  
96              html.write(maps.toString());
97  
98              return true;
99          } finally {
100             html.writeln("</a>");
101             writeFooter(html);
102         }
103     }
104 
105     private void writeHeader(Database db, String title, boolean hasImpliedRelationships, LineWriter html) throws IOException {
106         writeHeader(db, null, title, true, html);
107         html.writeln("<table class='container' width='100%'>");
108         html.writeln("<tr><td class='container'>");
109         writeGeneratedBy(db.getConnectTime(), html);
110         html.writeln("</td>");
111         html.writeln("<td class='container' align='right' valign='top' rowspan='2'>");
112         writeLegend(false, html);
113         html.writeln("</td></tr>");
114         html.writeln("<tr><td class='container' align='left' valign='top'>");
115         if (hasImpliedRelationships) {
116             html.writeln("<form action=''>");
117             html.writeln(" <label for='removeImpliedOrphans'><input type=checkbox id='removeImpliedOrphans'>");
118             html.writeln("  Hide tables with implied relationships</label>");
119             html.writeln("</form>");
120         }
121         html.writeln("</td></tr></table>");
122     }
