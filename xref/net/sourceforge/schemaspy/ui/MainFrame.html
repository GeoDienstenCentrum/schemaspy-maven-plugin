36  public class MainFrame extends JFrame {
37      private static final long serialVersionUID = 1L;
38      private JPanel jContentPane = null;

40      private JPanel buttonBar = null;
41      private JButton launchButton = null;
42      private JPanel header;

44      /**
45       * This is the default constructor
46       */
47      public MainFrame() {
48          super();
49          initialize();
50      }

52      /**
53       * This method initializes this
54       *
55       * @return void
56       */
57      private void initialize() {
58          setContentPane(getJContentPane());
59          setTitle("SchemaSpy");
60          this.setSize(new Dimension(500, 312));
61          addWindowListener(new WindowAdapter() {
62              @Override
63              public void windowClosing(WindowEvent e) {
64                  System.exit(0);
65              }
66          });
67      }
68  
69      /**
70       * This method initializes dbConfigPanel
71       *
72       * @return javax.swing.JPanel
73       */
74      private JPanel getDbConfigPanel() {
75          if (dbConfigPanel == null) {
76              dbConfigPanel = new DbConfigPanel();
77          }
78          return dbConfigPanel;
79      }
80  
81      private JPanel getJContentPane() {
82          if (jContentPane == null) {
83              jContentPane = new JPanel();
84              jContentPane.setLayout(new GridBagLayout());
85              GridBagConstraints constraints = new GridBagConstraints();
86              constraints.gridx = 0;
87              constraints.gridy = GridBagConstraints.RELATIVE;
88              constraints.weightx = 1.0;
89  
90              constraints.anchor = GridBagConstraints.CENTER;
91              constraints.insets = new Insets(4, 0, 4, 0);
92              jContentPane.add(getHeaderPanel(), constraints);
93              constraints.insets = new Insets(0, 0, 0, 0);
94  
95              constraints.fill = GridBagConstraints.BOTH;
96              constraints.anchor = GridBagConstraints.NORTHWEST;
97              constraints.weighty = 1.0;
98              //JScrollPane scroller = new JScrollPane();
99              //scroller.setBorder(null);
100             //scroller.setViewportView(getDbConfigPanel());
101             //scroller.setViewportBorder(new BevelBorder(BevelBorder.LOWERED));
102             //jContentPane.add(scroller, constraints);
103             jContentPane.add(getDbConfigPanel(), constraints);
104 
105 //            constraints.fill = GridBagConstraints.VERTICAL;
106 //            constraints.weighty = 0.0;
107 //            JLabel filler = new JLabel();
108 //            filler.setPreferredSize(new Dimension(0, 0));
109 //            filler.setMinimumSize(new Dimension(0, 0));
110 //            jContentPane.add(filler, constraints);
111 
112             constraints.anchor = GridBagConstraints.SOUTHEAST;
113             constraints.fill = GridBagConstraints.NONE;
114             constraints.weighty = 0.0;
115             jContentPane.add(getButtonBar(), constraints);
116         }
117         return jContentPane;
118     }
119 
120     /**
121      * This method initializes buttonBar
122      *
123      * @return javax.swing.JPanel
124      */
125     private JPanel getButtonBar() {
126         if (buttonBar == null) {
127             buttonBar = new JPanel();
128             buttonBar.setLayout(new FlowLayout(FlowLayout.TRAILING));
129             buttonBar.add(getLaunchButton(), null);
130         }
131         return buttonBar;
132     }
133 
134     private JPanel getHeaderPanel() {
135         if (header == null) {
136             header = new JPanel();
137             header.setLayout(new GridBagLayout());
138             GridBagConstraints constraints = new GridBagConstraints();
139             constraints.gridx = 0;
140             constraints.gridy = 0;
141             header.add(new JLabel("SchemaSpy - Graphical Database Metadata Browser"), constraints);
142             constraints.gridx = 0;
143             constraints.gridy++;
144             header.add(new JLabel("Select a database type and fill in the required fields"), constraints);
145         }
146         return header;
147     }
148 
149     /**
150      * This method initializes launchButton
151      *
152      * @return javax.swing.JButton
153      */
154     private JButton getLaunchButton() {
155         if (launchButton == null) {
156             launchButton = new JButton();
157             launchButton.setText("Launch");
158         }
159         return launchButton;
160     }
